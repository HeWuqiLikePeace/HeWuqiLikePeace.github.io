原理就是利用9号接口的PWM功能调整其占空比来实现亮度的变化

接线部分：正极接9号接口（或在代码中修改为3、5、6、9、10、11号）

                    负极接GND即可

使用USB接口5V供电或DC接口7-12V供电

void fadeOff(int time) {
for (int i = 255; i >= 0; i = i + (-5)) {
analogWrite(9,i);
delay((time / (255 / 5)));
}
}

void fadeOn(int time) {
for (int i = 0; i <= 255; i = i + (5)) {
analogWrite(9,i);
delay((time / (255 / 5)));
}
}

}

void loop(){
fadeOn(2000);
fadeOff(2000);

}
